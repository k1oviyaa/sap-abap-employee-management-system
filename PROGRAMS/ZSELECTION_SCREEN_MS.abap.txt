REPORT ZSELECTION_SCREEN_MS.

PARAMETERS: p_dept TYPE ZOV_DEPT_E.

DATA: it_emp TYPE TABLE OF zemployeems,
      wa_emp TYPE zemployeems.

SELECT * FROM zemployeems
  INTO TABLE it_emp
  WHERE DEPARTMENT = p_dept.

IF it_emp IS INITIAL.
  WRITE: / 'No employees found in this department.'.
ELSE.
  LOOP AT it_emp INTO wa_emp.
    WRITE: / wa_emp-emp_id,
             wa_emp-emp_name,
             wa_emp-designation,
             wa_emp-salary.
  ENDLOOP.
ENDIF.
